```python
def get_real_stock_data(stock_code, days=500):
    import futu as ft
    quote_ctx = ft.OpenQuoteContext(host='127.0.0.1', port=11111)
    ret, data = quote_ctx.get_history_kline(
        f"HK.{stock_code}", 
        start=(datetime.now() - timedelta(days=days*2)).strftime('%Y-%m-%d'),
        end=datetime.now().strftime('%Y-%m-%d'),
        ktype=ft.KLType.K_DAY
    )
    quote_ctx.close()
    return data

# 2. 實時預測系統
class RealTimePredictor:
    def __init__(self, stock_codes):
        self.models = {}
        for code in stock_codes:
            self.models[code] = self.train_model_for_stock(code)
    
    def predict_all(self):
        predictions = {}
        for code, model in self.models.items():
            data = get_real_stock_data(code, days=100)
            prediction = model.predict(data)
            predictions[code] = prediction
        return predictions
```
